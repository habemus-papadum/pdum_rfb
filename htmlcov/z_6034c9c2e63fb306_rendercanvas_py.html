1"""A `rendercanvas <https://rendercanvas.readthedocs.io>`_ backend that streams over ``pdum.rfb``. 

2 

3``rendercanvas`` is the canvas-abstraction layer under ``wgpu`` / ``pygfx`` / 

4``fastplotlib``: a render engine targets an abstract canvas and a *backend* decides where 

5the pixels go (a glfw window, a Qt widget, an offscreen buffer, a notebook...). This 

6module is a backend whose pixels go to a :class:`pdum.rfb.Display` — i.e. a ``wgpu``/ 

7``pygfx`` app renders unchanged and the result streams to the browser over this library's 

8WebSocket + Web-Worker pipeline (image or H.264/WebCodecs), with input flowing back. 

9 

10It is the spiritual equivalent of ``rendercanvas``'s own ``jupyter_rfb`` backend, but on 

11this library's transport. **Cross-platform**: the ``"bitmap"`` present method downloads 

12the rendered frame to a host ``numpy`` array, so it works identically on macOS and Linux 

13(no CUDA/NVENC required). The GPU zero-copy path is a separate, Linux-only future track 

14(see ``docs/wgpu_nvenc_zerocopy.md``). 

15 

16Usage (own your asyncio loop; reuse :func:`pdum.rfb.serve`):: 

17 

18 import asyncio, pdum.rfb as rfb 

19 from pdum.rfb.rendercanvas import RfbRenderCanvas, loop 

20 import pygfx 

21 

22 async def main(): 

23 display = await rfb.serve(1280, 720, port=8765) 

24 canvas = RfbRenderCanvas(display=display, size=(1280, 720)) 

25 renderer = pygfx.renderers.WgpuRenderer(canvas) 

26 scene, camera = build_scene() # your pygfx scene 

27 controller = pygfx.OrbitController(camera, register_events=renderer) 

28 

29 def animate(): 

30 renderer.render(scene, camera) 

31 canvas.request_draw(animate) 

32 

33 canvas.request_draw(animate) 

34 try: 

35 await loop.run_async() # runs on the current asyncio loop 

36 finally: 

37 await display.aclose() 

38 

39 asyncio.run(main()) 

40 

41Notes 

42----- 

43* Browser input (pointer / wheel / key) is drained from the display and delivered to the 

44 canvas event system, so ``pygfx`` controllers (orbit camera, etc.) work. When you use 

45 this backend the events go to the **canvas** (``canvas.add_event_handler`` / controllers), 

46 *not* to ``display.poll_events()`` — the backend drains that queue for you. 

47* The canvas size (set at construction or via ``set_logical_size``) is the render 

48 resolution and what gets published. Browser ``resize`` is informational here (the 

49 publisher owns the resolution), so it is *not* auto-applied — matching the shared-display 

50 model. Keep the size **even** for the H.264 path. 

51* Event-schema note: this library emits the `renderview <https://github.com/pygfx/renderview>`_ 

52 vocabulary (``type``/``timestamp``); ``rendercanvas`` 2.x still consumes the legacy keys 

53 (``event_type``/``time_stamp``). :func:`_to_rendercanvas_event` renames them — the 

54 *values* (logical coords, ``1=left/2=right/3=middle`` buttons, tuple ``buttons``, 

55 capitalized ``modifiers``) already match, so it is a pure key-rename. When ``rendercanvas`` 

56 adopts ``type`` it collapses to the identity. 

57"""

80#: Browser event types we forward to the canvas. ``resize``/``set_viewport`` are skipped 

81#: (the canvas owns its render size); the rest of the renderview vocabulary maps 1:1. 

82_FORWARD = frozenset({"pointer_move", "pointer_down", "pointer_up", "wheel", "key_down", "key_up"})

85def _to_rendercanvas_event(event: dict) -> dict | None: 

86 """Translate a renderview-shaped ``pdum.rfb`` event to a ``rendercanvas`` event. 

87 

88 Renames ``type`` → ``event_type`` and ``timestamp`` → ``time_stamp`` (the keys 

89 ``rendercanvas`` 2.x expects); every value (coords, ``button``/``buttons``, 

90 ``modifiers``) is already in the shared renderview convention. Returns ``None`` for 

91 events that should not be forwarded (e.g. ``resize``). 

92 """ 

93 etype = event.get("type") 

94 if etype not in _FORWARD: 

95 return None 

96 out = dict(event) 

97 out["event_type"] = etype 

98 out.pop("type", None) 

99 ts = out.pop("timestamp", None) 

100 if ts is not None: 

101 out["time_stamp"] = ts 

102 return out

109class RfbRenderCanvas(BaseRenderCanvas): 

110 """A ``rendercanvas`` backend that publishes each rendered frame to a :class:`~pdum.rfb.display.Display`. 

111 

112 Parameters 

113 ---------- 

114 display: 

115 A started :class:`~pdum.rfb.display.Display` (from ``await pdum.rfb.serve(...)``). 

116 Rendered frames are published to it and browser input is drained from it. 

117 size: 

118 Logical canvas size ``(width, height)`` — the render resolution and the published 

119 frame size. Defaults to the display's current size. 

120 **kwargs: 

121 Forwarded to :class:`rendercanvas.base.BaseRenderCanvas` (``update_mode``, 

122 ``max_fps``, ``title``, ...). 

123 """

143 # `data` is a contiguous (H, W, 4) uint8 RGBA array (wgpu downloaded it from the 

144 # render texture). publish() tags (H, W, 4) as rgba8 and fans it out to viewers. 

145 if self._closed or self._display._closed:

161 def _rc_gui_poll(self) -> None: 

162 # Called regularly by the scheduler. Drain browser input from all viewers and

164 for ev in self._display.poll_events(): 

165 rc_event = _to_rendercanvas_event(ev.event)
