1"""PyAV-free GPU H.264 encoder via the NVENC SDK (``pdum.nvenc``). 

2 

3The third GPU path, and the only **PyAV-free** one. Where 

4:class:`~pdum.rfb.encoders.nvenc.NvencCpuEncoder` uploads a host ``rgb24`` frame and 

5:class:`~pdum.rfb.encoders.nvenc_gpu_pyav.NvencGpuPyavEncoder` needs **PyAV ≥ 18**, this 

6backend hands a GPU-resident NV12 buffer straight to NVIDIA's ``NvEncoderCuda`` (the 

7``habemus-papadum-nvenc`` package, ``import pdum.nvenc``) and gets **Annex B** back — 

8no ffmpeg/PyAV anywhere in the encode path. It is the fastest path measured on the 

9test hardware (see ``docs/performance.md``). 

10 

11Input handling mirrors the zero-copy CUDA backend: 

12 

13* a CUDA ``nv12`` frame — encoded as-is (the true zero-copy case); 

14* a CUDA ``rgb24``/``rgba8`` frame — converted to NV12 on the GPU first 

15 (:func:`pdum.rfb.gpu.rgb_to_nv12`); 

16* a host ``rgb24``/``rgba8`` frame — uploaded then converted (graceful fallback). 

17 

18Gate on :func:`nvenc_gpu_pdum_available` before constructing one. Emits the same 

19low-latency Annex B as the other H.264 backends, so the wire format, forced-keyframe 

20handling, and payload packing are unchanged. Fixed-resolution: a resize rebuilds the 

21encoder (the session does this via its ``encoder_factory``) and forces a keyframe. 

22"""

34def _contains_idr(annexb: bytes) -> bool: 

35 """True if the Annex B buffer carries an IDR slice (NAL unit type 5). 

36 

37 Start-code emulation prevention guarantees ``00 00 01`` never appears inside a 

38 NAL payload, so splitting on it yields exactly the NAL boundaries; the byte after 

39 each boundary is the NAL header whose low 5 bits are the unit type. 

40 """ 

41 for nal in annexb.split(b"\x00\x00\x01")[1:]: 

42 if nal and (nal[0] & 0x1F) == 5: 

43 return True 

44 return False

47class NvencGpuPdumEncoder: 

48 """Encode CUDA (or host, with upload) frames to H.264 Annex B via the NVENC SDK. 

49 

50 Same constructor / :class:`~pdum.rfb.types.EncoderBackend` interface as the other 

51 H.264 backends. Frames narrower than 

52 :data:`~pdum.rfb.encoders.nvenc.NVENC_MIN_WIDTH` are rejected (NVENC cannot open 

53 below it) and dimensions must be even (NV12). 

54 """ 

55 

56 encoder_label = "nvenc-gpu-pdum" 

57 

58 def __init__( 

59 self, 

60 *, 

61 width: int, 

62 height: int, 

63 fps: int = 30, 

64 bitrate: int = 12_000_000, 

65 codec_string: str | None = None, 

66 preset: str = "p4", 

67 tuning: str = "ll", 

68 ) -> None: 

69 if width < NVENC_MIN_WIDTH: 

70 raise ValueError( 

71 f"NVENC requires width >= {NVENC_MIN_WIDTH}; got {width}. " 

72 "Use a larger framebuffer or fall back to the libx264 encoder." 

73 ) 

74 if width % 2 or height % 2: 

75 raise ValueError(f"NV12 requires even dimensions; got {width}x{height}") 

76 

77 import cupy as cp 

78 from pdum.nvenc import NvencEncoder 

79 

80 self.width = width 

81 self.height = height 

82 self.fps = fps 

83 self.bitrate = bitrate 

84 self.codec_string = codec_string or DEFAULT_H264_CODEC 

85 self.frame_index = 0 

86 self._duration_us = int(1_000_000 / fps) 

87 # Reusable NV12 staging buffer for the rgb/host input paths. ll tuning with no 

88 # lookahead consumes each frame before the next, so a single buffer is safe. 

89 self._nv12 = cp.empty((height + height // 2, width), cp.uint8) 

90 # cuda_context=0 -> retain the device *primary* context (the one CuPy uses), so 

91 # CuPy device pointers are valid to NVENC with no cross-context copy. 

92 self._enc = NvencEncoder( 

93 width, 

94 height, 

95 codec="h264", 

96 preset=preset, 

97 tuning=tuning, 

98 fps=fps, 

99 gop=fps, 

100 bitrate=bitrate, 

101 )

103 def _packed_nv12(self, frame: RawFrame): 

104 """Return a contiguous CUDA NV12 ``(H+H//2, W)`` buffer for ``frame``.""" 

105 import cupy as cp 

106 

107 from ..gpu import rgb_to_nv12 

108 

109 if frame.memory == "cuda" and frame.pixel_format == "nv12": 

110 packed = cp.ascontiguousarray(cp.asarray(frame.data)) 

111 if packed.shape != self._nv12.shape: 

112 raise ValueError(f"nv12 frame shape {packed.shape!r} != encoder {self._nv12.shape!r}") 

113 return packed 

114 if frame.pixel_format not in ("rgb24", "rgba8"): 

115 raise TypeError(f"NvencGpuPdumEncoder cannot encode {frame.pixel_format!r} frames") 

116 # rgb24/rgba8, CUDA or host: convert (uploading first if host) into the buffer. 

117 return rgb_to_nv12(frame.data, out=self._nv12) 

118 

119 def encode(self, frame: RawFrame, *, force_keyframe: bool = False) -> list[EncodedPayload]: 

120 import cupy as cp 

121 

122 packed = self._packed_nv12(frame) 

123 # Ensure the NV12 (kernel/upload above) is complete before NVENC's intra-GPU 

124 # copy reads it. Sub-ms at these sizes; keeps the path correct across streams. 

125 cp.cuda.runtime.deviceSynchronize() 

126 data = self._enc.encode(packed, force_idr=force_keyframe) 

127 self.frame_index += 1 

128 if not data: 

129 return [] 

130 keyframe = force_keyframe or _contains_idr(data) 

131 return [self._payload(frame.seq, frame.timestamp_us, data, keyframe)] 

132 

133 def encode_still(self, frame: RawFrame) -> list[EncodedPayload]: 

134 """Settled-scene still: a forced **IDR** of the resting frame (see the CPU 

135 backend's :meth:`~pdum.rfb.encoders.h264_cpu.H264CpuEncoder.encode_still`).""" 

136 return self.encode(frame, force_keyframe=True) 

137 

138 def flush(self) -> list[EncodedPayload]: 

139 data = self._enc.flush() 

140 if not data: 

141 return [] 

142 return [self._payload(-1, 0, data, _contains_idr(data))] 

143 

144 def close(self) -> None: 

145 try: 

146 self._enc.close() 

147 except Exception: # pragma: no cover - encoder may already be closed 

148 pass

153 kind="video",

155 width=self.width,

159 keyframe=keyframe, 

160 duration_us=self._duration_us, 

161 metadata={"bitstream": "annexb", "encoder": self.encoder_label}, 

162 )

165@functools.lru_cache(maxsize=1) 

166def nvenc_gpu_pdum_available() -> bool: 

167 """True if the PyAV-free SDK NVENC path is usable in this process (cached). 

168 

169 Checks CuPy + ``pdum.nvenc`` importable, then actually opens an NVENC session and 

170 encodes two frames (no decode, so it stays PyAV-free). Runs at most once per 

171 process because it opens a real encoder session. 

172 """ 

173 if importlib.util.find_spec("cupy") is None or importlib.util.find_spec("pdum.nvenc") is None: 

174 return False 

175 try: 

176 import cupy as cp 

177 

178 from ..gpu import cuda_frame 

179 

180 enc = NvencGpuPdumEncoder(width=256, height=128, fps=4, bitrate=2_000_000) 

181 total = 0 

182 for seq in range(2): 

183 nv12 = cp.zeros((128 + 64, 256), cp.uint8) 

184 frame = cuda_frame(nv12, pixel_format="nv12", height=128, seq=seq) 

185 total += sum(len(p.payload) for p in enc.encode(frame, force_keyframe=(seq == 0))) 

186 total += sum(len(p.payload) for p in enc.flush()) 

187 enc.close() 

188 return total > 0 

189 except Exception: 

190 return False

193def self_test(width: int = 256, height: int = 256, frames: int = 8) -> bool: 

194 """Encode synthetic CUDA NV12 frames via the SDK and decode them back (needs PyAV).""" 

195 if not nvenc_gpu_pdum_available(): 

196 return False 

197 

198 import cupy as cp 

199 

200 from ..gpu import cuda_frame 

201 from ..testing import decode_annexb 

202 

203 width = max(width, NVENC_MIN_WIDTH) 

204 enc = NvencGpuPdumEncoder(width=width, height=height, fps=int(frames)) 

205 chunks: list[bytes] = [] 

206 for seq in range(frames): 

207 nv12 = cp.empty((height + height // 2, width), cp.uint8) 

208 nv12[:height] = (seq * 7) % 256 # moving luma 

209 nv12[height:] = 128 # neutral chroma 

210 frame = cuda_frame(nv12, pixel_format="nv12", height=height, seq=seq) 

211 for payload in enc.encode(frame, force_keyframe=(seq == 0)): 

212 chunks.append(payload.payload) 

213 for payload in enc.flush(): 

214 chunks.append(payload.payload) 

215 enc.close() 

216 

217 decoded = decode_annexb(b"".join(chunks)) 

218 if not decoded: 

219 return False 

220 return all(f.width == width and f.height == height for f in decoded)
