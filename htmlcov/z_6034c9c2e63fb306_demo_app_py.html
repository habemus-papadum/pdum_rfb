1"""The Textual control panel for ``pdum-rfb demo`` (imported lazily; needs ``[demo]``). 

2 

3A two-column TUI: pick a **demo scene** and an **encode backend** from lists, retune 

4**bitrate/fps**, and watch **live per-session stats**. Selecting a scene swaps what the 

5render loop publishes; selecting a backend calls 

6:meth:`pdum.rfb.server._StreamHost.switch_backend`, which reconfigures every live viewer 

7on the same WebSocket (no reconnect). 

8""" 

9

12from typing import Any

15from textual.containers import Horizontal, Vertical 

16from textual.widgets import Button, Footer, Header, Input, OptionList, RichLog, Static 

17from textual.widgets.option_list import Option 

18 

19from .demo_tui import _parse_bitrate

22class DemoApp(App): 

23 """Drive a live framebuffer demo: switch scenes & encode backends, retune, watch stats."""

46 demos: list[Any],

64 def compose(self) -> ComposeResult: 

65 yield Header(show_clock=True) 

66 with Horizontal(id="cols"): 

67 with Vertical(id="left"): 

68 demo_list = OptionList( 

69 *[Option(f"{d.name} — {d.description}", id=d.key) for d in self.demos], id="demos" 

70 ) 

71 demo_list.border_title = "Demo scene" 

72 yield demo_list 

73 backend_list = OptionList(*[Option(label, id=bid) for bid, label in self.backends], id="backends") 

74 backend_list.border_title = "Encode backend (switches live)" 

75 yield backend_list 

76 with Vertical(id="quality"): 

77 yield Static("Quality") 

78 with Horizontal(): 

79 yield Input(value=_fmt_bitrate(self.bitrate), id="bitrate", placeholder="8M") 

80 yield Input(value=str(self.fps), id="fps", placeholder="30") 

81 yield Button("Apply", id="apply", variant="primary") 

82 with Vertical(id="right"): 

83 url = Static( 

84 f"Open in your browser:\n[b link={self.url}]{self.url}[/]\nVite: {self.vite_status}", id="url" 

85 ) 

86 url.border_title = "Browser client" 

87 stats = Static("", id="stats") 

88 stats.border_title = "Live stats" 

89 yield stats 

90 log = RichLog(id="log", markup=True, highlight=True) 

91 log.border_title = "Log"

95 def on_mount(self) -> None: 

96 self.title = "pdum-rfb demo" 

97 self.sub_title = self.url 

98 log = self.query_one("#log", RichLog) 

99 log.write(f"[b]Client URL:[/] {self.url}") 

100 log.write(f"[b]Vite:[/] {self.vite_status}") 

101 log.write(f"[b]WebSocket:[/] ws://{self.rfb_display.port and ''}… (port {self.rfb_display.port})") 

102 log.write("Pick a [b]demo scene[/] and an [b]encode backend[/]; both apply live.") 

103 # Reflect the starting scene selection. 

104 self.query_one("#demos", OptionList).highlighted = 0 

105 self.set_interval(1.0, self._refresh_stats) 

106 self._refresh_stats() 

107 

108 def on_option_list_option_selected(self, event: OptionList.OptionSelected) -> None: 

109 which = event.option_list.id 

110 oid = event.option.id 

111 log = self.query_one("#log", RichLog) 

112 if which == "demos": 

113 self.state.select(oid) 

114 log.write(f"[cyan]scene →[/] {oid}") 

115 elif which == "backends": 

116 try: 

117 self.rfb_host.switch_backend(oid) 

118 log.write(f"[green]backend →[/] {oid} (live switch; browser follows on next keyframe)") 

119 except Exception as exc: # noqa: BLE001 

120 log.write(f"[red]backend switch failed:[/] {exc}") 

121 self._refresh_stats() 

122 

123 def on_button_pressed(self, event: Button.Pressed) -> None: 

124 if event.button.id != "apply": 

125 return 

126 log = self.query_one("#log", RichLog) 

127 try: 

128 br = _parse_bitrate(self.query_one("#bitrate", Input).value) 

129 fps = int(self.query_one("#fps", Input).value) 

130 self.rfb_host.set_quality(bitrate=br, fps=fps) 

131 self.bitrate, self.fps = br, fps 

132 log.write(f"[green]quality →[/] bitrate={br / 1e6:.1f}M fps={fps} (encoders rebuilt)") 

133 except Exception as exc: # noqa: BLE001 

134 log.write(f"[red]quality apply failed:[/] {exc}") 

135 self._refresh_stats() 

136 

137 def _current_backend(self) -> str: 

138 h = self.rfb_host 

139 if h._force_transport == "image": 

140 return f"image:{h.image_mode}" 

141 if h._force_transport == "h264": 

142 return h.video_encoder 

143 return f"auto ({h.video_encoder})" 

144 

145 def _refresh_stats(self) -> None: 

146 metrics = self.rfb_host.metrics() 

147 lines = [ 

148 f"scene: {self.state.active_key}", 

149 f"backend: {self._current_backend()}", 

150 f"clients: {self.rfb_display.client_count}", 

151 ] 

152 if self.state.last_error: 

153 lines.append(f"[red]scene error: {self.state.last_error}[/]") 

154 for i, m in enumerate(metrics): 

155 lines.append( 

156 f" · c{i}: {m['fps_sent']:.1f} fps {m['bitrate_bps'] / 1e6:.1f} Mbps " 

157 f"enc {m['encode_ms']:.1f}ms rtt {m['rtt_ms']:.0f}ms " 

158 f"inflight {m['inflight']} dropped {m['frames_dropped']} q {m['decode_queue_size']}" 

159 ) 

160 if not metrics: 

161 lines.append(" (no viewers connected yet — open the URL above)") 

162 self.query_one("#stats", Static).update("\n".join(lines))

165def _fmt_bitrate(bps: int) -> str: 

166 return f"{bps / 1e6:.1f}M" if bps >= 1_000_000 else f"{bps // 1000}k"
