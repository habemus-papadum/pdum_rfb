1"""Hardware H.264 encoder via NVIDIA NVENC (the roadmap's GPU backend). 

2 

3This rides on **PyAV's bundled ffmpeg** rather than NVIDIA's ``PyNvVideoCodec``: 

4the ``av`` wheel ships an ffmpeg built with ``--enable-nvenc``, exposing the 

5``h264_nvenc`` encoder, which ``dlopen``\\s ``libnvidia-encode`` at runtime. So 

6this backend needs no Python dependency beyond ``av`` (already the ``h264`` / 

7``nvenc`` extra) — only a host NVIDIA driver + an NVENC-capable GPU, which pip 

8cannot install. 

9 

10Like :class:`~pdum.rfb.encoders.h264_cpu.H264CpuEncoder` it emits **Annex B** 

11access units (in-band SPS/PPS on key frames) configured for low latency 

12(``preset=p4``/``tune=ll``, no B-frames, ~1 s forced-IDR cadence). It subclasses 

13the libx264 encoder and only swaps the underlying ``av.CodecContext``, so the 

14frame conversion, forced-keyframe handling, and payload packing are shared. 

15 

16Availability is gated by :func:`nvenc_cpu_available`, which checks the OS, that PyAV 

17exposes ``h264_nvenc``, and that the encoder actually opens on the GPU. 

18""" 

19 

20from __future__ import annotations 

21 

22import sys 

23import time 

24from fractions import Fraction 

25 

26from ..protocol import DEFAULT_H264_CODEC 

27from .h264_cpu import H264CpuEncoder, h264_available 

28 

29#: NVENC H.264 has a hardware minimum frame width (160 on the tested Ada GPU; 

30#: widths <160 fail ``avcodec_open2`` with EINVAL). Height may be smaller. 

31NVENC_MIN_WIDTH = 160 

32 

33#: OSes where NVENC can exist. macOS has no NVIDIA driver / NVENC. 

34_NVENC_PLATFORMS = ("linux", "win32", "cygwin") 

35 

36#: Codec name in PyAV/ffmpeg for the NVENC H.264 encoder. 

37_NVENC_CODEC = "h264_nvenc" 

38 

39# Cache the probe result: opening an NVENC session is not free and consumer GPUs 

40# cap concurrent sessions, so we never want to probe more than once per process. 

41_nvenc_ok: bool | None = None 

42

44def nvenc_codec_available() -> bool: 

45 """True if PyAV is importable and lists the ``h264_nvenc`` encoder. 

46 

47 This is a cheap, side-effect-free check (it does not touch the GPU). It is 

48 necessary but not sufficient — the encoder still has to *open* on a real 

49 device, which :func:`nvenc_cpu_available` verifies. 

50 """ 

51 if sys.platform not in _NVENC_PLATFORMS: 

52 return False 

53 if not h264_available(): 

54 return False 

55 try: 

56 import av 

57 

58 return _NVENC_CODEC in av.codecs_available 

59 except Exception: # pragma: no cover - defensive 

60 return False

63def _probe_open(retries: int = 3) -> bool: 

64 """Actually open + encode one frame on the GPU to prove NVENC is usable. 

65 

66 Consumer GPUs cap concurrent NVENC sessions and rapid open/close churn can 

67 transiently return EINVAL, so the probe retries a few times before giving up. 

68 """ 

69 import av 

70 import numpy as np 

71 

72 w = h = 256 # comfortably above NVENC_MIN_WIDTH 

73 for _attempt in range(retries): 

74 try: 

75 ctx = av.CodecContext.create(_NVENC_CODEC, "w") 

76 ctx.width = w 

77 ctx.height = h 

78 ctx.pix_fmt = "yuv420p" 

79 ctx.time_base = Fraction(1, 30) 

80 ctx.framerate = Fraction(30, 1) 

81 ctx.bit_rate = 2_000_000 

82 vf = av.VideoFrame.from_ndarray(np.zeros((h, w, 3), dtype=np.uint8), format="rgb24") 

83 vf = vf.reformat(format="yuv420p") 

84 vf.pts = 0 

85 vf.time_base = Fraction(1, 30) 

86 list(ctx.encode(vf)) 

87 list(ctx.encode(None)) # flush; PyAV CodecContext has no close() 

88 return True 

89 except Exception: # pragma: no cover - hardware/driver dependent 

90 time.sleep(0.25) 

91 return False

94def nvenc_cpu_available() -> bool: 

95 """True if a usable NVENC H.264 encoder is present (cached). 

96 

97 Guards, in order: the OS must be one where NVENC exists (not macOS); PyAV 

98 must expose ``h264_nvenc``; and the encoder must actually open and encode a 

99 frame on the GPU (which requires an NVENC-capable device and a working 

100 driver). The result is cached for the lifetime of the process. 

101 """ 

102 global _nvenc_ok 

103 if _nvenc_ok is not None: 

104 return _nvenc_ok 

105 if not nvenc_codec_available(): 

106 _nvenc_ok = False 

107 return _nvenc_ok 

108 _nvenc_ok = _probe_open() 

109 return _nvenc_ok

112class NvencCpuEncoder(H264CpuEncoder): 

113 """Encode CPU ``rgb24`` frames to H.264 Annex B on the GPU via NVENC. 

114 

115 Drop-in for :class:`H264CpuEncoder` (same constructor and ``EncoderBackend`` 

116 interface); only the underlying codec context differs. Frames narrower than 

117 :data:`NVENC_MIN_WIDTH` are rejected because NVENC cannot open below it. 

118 """ 

119 

120 encoder_label = "nvenc-cpu" 

121 

122 def __init__( 

123 self, 

124 *, 

125 width: int, 

126 height: int, 

127 fps: int = 30, 

128 bitrate: int = 12_000_000, 

129 codec_string: str | None = None, 

130 ) -> None: 

131 if width < NVENC_MIN_WIDTH: 

132 raise ValueError( 

133 f"NVENC requires width >= {NVENC_MIN_WIDTH}; got {width}. " 

134 "Use a larger framebuffer or fall back to the libx264 encoder." 

135 ) 

136 super().__init__( 

137 width=width, 

138 height=height, 

139 fps=fps, 

140 bitrate=bitrate, 

141 codec_string=codec_string or DEFAULT_H264_CODEC, 

142 ) 

143 

144 def _make_context(self): 

145 import av 

146 

147 ctx = av.CodecContext.create(_NVENC_CODEC, "w") 

148 ctx.width = self.width 

149 ctx.height = self.height 

150 ctx.pix_fmt = "yuv420p" 

151 ctx.time_base = Fraction(1, self.fps) 

152 ctx.framerate = Fraction(self.fps, 1) 

153 ctx.bit_rate = self.bitrate 

154 # Low latency: fast preset, low-latency tune, no B-frames, immediate 

155 # output, ~1 s forced-IDR cadence. ``rc=vbr`` treats ``bit_rate`` as a 

156 # target/ceiling and lets the stream **undershoot** on static frames — 

157 # critical for sparse scientific scenes (CBR would pad to the full bitrate 

158 # even when nothing changed, ~60x more bytes for identical quality, and 

159 # mirrors how the libx264 ABR path already behaves). ``profile=baseline`` 

160 # makes the SPS advertise profile_idc 66 to match the negotiated 

161 # ``avc1.42E01F`` codec string (NVENC otherwise defaults to main/high). 

162 # NVENC emits Annex B with in-band SPS/PPS on every IDR for a raw stream. 

163 ctx.options = { 

164 "preset": "p4", 

165 "tune": "ll", 

166 "rc": "vbr", 

167 "profile": "baseline", 

168 "bf": "0", 

169 "delay": "0", 

170 "forced-idr": "1", 

171 "g": str(self.fps), 

172 } 

173 return ctx

176def self_test(width: int = 256, height: int = 256, frames: int = 8) -> bool: 

177 """Encode a few synthetic frames via NVENC and decode them back with PyAV. 

178 

179 Returns ``True`` if the produced Annex B bitstream decodes to a plausible 

180 number of frames at the expected resolution. Doubles as a runtime check that 

181 NVENC is actually usable end-to-end. 

182 """ 

183 if not nvenc_cpu_available(): 

184 return False 

185 

186 import numpy as np 

187 

188 from ..testing import decode_annexb 

189 from ..types import RawFrame 

190 

191 width = max(width, NVENC_MIN_WIDTH) 

192 enc = NvencCpuEncoder(width=width, height=height, fps=int(frames)) 

193 chunks: list[bytes] = [] 

194 for seq in range(frames): 

195 arr = np.full((height, width, 3), (seq * 7) % 256, dtype=np.uint8) 

196 arr[:, : width // 2] = ((seq * 11) % 256, (seq * 5) % 256, 64) 

197 for payload in enc.encode( 

198 RawFrame(seq, width, height, seq * 1000, "rgb24", "cpu", arr), 

199 force_keyframe=(seq == 0), 

200 ): 

201 chunks.append(payload.payload) 

202 for payload in enc.flush(): 

203 chunks.append(payload.payload) 

204 enc.close() 

205 

206 decoded = decode_annexb(b"".join(chunks)) 

207 if not decoded: 

208 return False 

209 return all(f.width == width and f.height == height for f in decoded)
