1"""GPU zero-copy helpers for the NVENC CUDA path (DLPack in, no host copy). 

2 

3This module lets a CUDA-resident frame (a CuPy / PyTorch / any ``__dlpack__`` or 

4``__cuda_array_interface__`` tensor) be encoded by NVENC **without a round-trip 

5through host memory** — the encoder reads the device buffer directly. It is the 

6zero-copy counterpart to :mod:`pdum.rfb.encoders.nvenc_cpu` (which uploads host 

7``rgb24`` and reformats to ``yuv420p`` on the CPU first). 

8 

9Everything here lazy-imports CuPy, so ``import pdum.rfb.gpu`` is always safe; the 

10functions raise only when actually called without CuPy. 

11 

12Requirements (see :func:`cuda_zerocopy_available`) 

13-------------------------------------------------- 

141. **CuPy** (``cupy-cuda13x`` / ``cupy-cuda12x``; cp314 wheels exist). 

152. An **NVENC-capable GPU + driver** (same gate as the host NVENC backend). 

163. **PyAV that can *encode* CUDA frames.** ``from_dlpack`` (frame *creation*) 

17 landed in PyAV 17.0, but feeding those frames to an encoder 

18 (``hw_frames_ctx`` adopted before ``avcodec_open2``) lands only in **PyAV 

19 18.0** (unreleased as of this writing; the fix is on ``main`` — issue #2199). 

20 On PyAV 17.x this raises *"hw_frames_ctx must be set when using GPU frames as 

21 input"*; there is **no pure-Python workaround** (PyAV exposes no handle to set 

22 ``avctx->hw_frames_ctx``), so a ``< 18`` install must build PyAV from source 

23 (``main`` or the small patch documented in ``docs/gpu_zerocopy.md``). 

24 

25Two non-obvious gotchas this module handles for you 

26--------------------------------------------------- 

27* **One shared CUDA context.** CuPy uses the device *primary* context; FFmpeg's 

28 CUDA hwcontext (``primary_ctx=True``) wants it created with 

29 ``CU_CTX_SCHED_BLOCKING_SYNC`` flags. If CuPy activates it first with the 

30 default (auto) flags, ``primary_ctx=True`` fails ("incompatible flags") and a 

31 separate (``primary_ctx=False``) context can't register CuPy's pointers 

32 ("resource register failed"). :func:`enable_cuda_context_sharing` pre-sets the 

33 flags — **call it before any CuPy/Torch CUDA work** (importing is fine; the 

34 first allocation/op is what activates the context). 

35* **NV12 must be one contiguous allocation** (Y plane then UV plane), because 

36 NVENC reads UV at ``base + pitch*height``. :func:`rgb_to_nv12` produces that 

37 layout; :func:`nv12_planes` slices it back into the two DLPack planes. 

38"""

68def enable_cuda_context_sharing(device_id: int = 0, *, sched: str = _FFMPEG_SCHED) -> bool: 

69 """Pre-set the device primary-context flags so CuPy and FFmpeg share one context. 

70 

71 **Call this once, first thing in your program, before any CuPy/PyTorch CUDA 

72 op.** It sets the device's primary-context scheduling flags to what FFmpeg's 

73 CUDA hwcontext wants (``CU_CTX_SCHED_BLOCKING_SYNC``); CuPy then retains that 

74 same primary context, and the zero-copy encoder (``primary_ctx=True``) can 

75 register CuPy's device pointers. 

76 

77 Returns ``True`` on success. Returns ``False`` (and changes nothing) if the 

78 CUDA driver can't be loaded. If the primary context is *already active* with 

79 different flags (e.g. CuPy already ran), the driver call may still succeed but 

80 the flags will not take effect until the context is reset — so order matters. 

81 

82 Parameters 

83 ---------- 

84 device_id: 

85 CUDA device ordinal. 

86 sched: 

87 One of ``"blocking_sync"`` (default, required for FFmpeg sharing), 

88 ``"spin"``, ``"yield"``, ``"auto"``. 

89 """ 

90 cu = _libcuda() 

91 if cu is None: 

92 return False 

93 flag = _SCHED_FLAGS[sched] 

94 if cu.cuInit(0) != 0: 

95 return False 

96 dev = ctypes.c_int() 

97 if cu.cuDeviceGet(ctypes.byref(dev), device_id) != 0: 

98 return False 

99 return cu.cuDevicePrimaryCtxSetFlags(dev, flag) == 0

104_RGB_TO_NV12_SRC = r""" 

105extern "C" __global__ void rgb_to_nv12(const unsigned char* rgb, unsigned char* out, int W, int H){ 

106 int x = blockIdx.x*blockDim.x + threadIdx.x; 

107 int yy = blockIdx.y*blockDim.y + threadIdx.y; 

108 if(x>=W || yy>=H) return; 

109 int i = (yy*W + x)*3; 

110 float R=rgb[i], G=rgb[i+1], B=rgb[i+2]; 

111 out[yy*W + x] = (unsigned char)(0.257f*R + 0.504f*G + 0.098f*B + 16.5f); // Y plane 

112 if(((x&1)==0) && ((yy&1)==0)){ // UV plane @ W*H 

113 int u = W*H + (yy>>1)*W + (x>>1)*2; 

114 out[u] = (unsigned char)(-0.148f*R - 0.291f*G + 0.439f*B + 128.5f); // Cb 

115 out[u+1] = (unsigned char)( 0.439f*R - 0.368f*G - 0.071f*B + 128.5f); // Cr 

116 } 

117} 

118""" 

119 

120_NV12_TO_RGB_SRC = r""" 

121extern "C" __global__ void nv12_to_rgb(const unsigned char* nv12, unsigned char* rgb, int W, int H){ 

122 int x = blockIdx.x*blockDim.x + threadIdx.x; 

123 int yy = blockIdx.y*blockDim.y + threadIdx.y; 

124 if(x>=W || yy>=H) return; 

125 float Y = nv12[yy*W + x]; 

126 int u = W*H + (yy>>1)*W + (x>>1)*2; 

127 float Cb = nv12[u], Cr = nv12[u+1]; 

128 float C = Y - 16.0f, D = Cb - 128.0f, E = Cr - 128.0f; 

129 int r = (int)(1.164f*C + 1.596f*E + 0.5f); 

130 int g = (int)(1.164f*C - 0.391f*D - 0.813f*E + 0.5f); 

131 int b = (int)(1.164f*C + 2.018f*D + 0.5f); 

132 int o = (yy*W + x)*3; 

133 rgb[o] = (unsigned char)(r<0?0:(r>255?255:r)); 

134 rgb[o+1] = (unsigned char)(g<0?0:(g>255?255:g)); 

135 rgb[o+2] = (unsigned char)(b<0?0:(b>255?255:b)); 

136} 

137"""

167def rgb_to_nv12(rgb: Any, *, out: Any | None = None): 

168 """Convert a device ``rgb24`` tensor ``(H, W, 3)`` to contiguous NV12. 

169 

170 Returns a CuPy ``uint8`` array of shape ``(H + H//2, W)``: the Y plane 

171 (``H`` rows) immediately followed by the interleaved UV plane (``H//2`` 

172 rows) — the single-allocation layout NVENC requires. Pass ``out`` to reuse a 

173 buffer across frames (the zero-copy encoder does this). 

174 """ 

175 import cupy as cp 

176 

177 rgb = _as_cupy(rgb) 

178 if rgb.ndim != 3 or rgb.shape[2] < 3: 

179 raise ValueError(f"rgb_to_nv12 expects (H, W, 3); got shape {rgb.shape!r}") 

180 h, w = int(rgb.shape[0]), int(rgb.shape[1]) 

181 if w % 2 or h % 2: 

182 raise ValueError(f"NV12 requires even dimensions; got {w}x{h}") 

183 if rgb.shape[2] != 3 or not rgb.flags.c_contiguous: 

184 rgb = cp.ascontiguousarray(rgb[:, :, :3]) 

185 if out is None: 

186 out = cp.empty((h + h // 2, w), cp.uint8) 

187 k_rgb2nv12, _, block = _kernels() 

188 k_rgb2nv12(_grid(w, h, block), block, (rgb, out, w, h)) 

189 return out

192def nv12_planes(packed: Any) -> tuple[Any, Any]: 

193 """Slice a contiguous NV12 ``(H + H//2, W)`` buffer into ``(Y, UV)`` planes. 

194 

195 The two returned CuPy arrays are *views* into ``packed`` (no copy), suitable 

196 for ``av.VideoFrame.from_dlpack([y, uv], format="nv12", ...)``. 

197 """ 

198 packed = _as_cupy(packed) 

199 rows = int(packed.shape[0]) 

200 h = rows * 2 // 3 

201 if h + h // 2 != rows: 

202 raise ValueError(f"not an NV12 buffer: {rows} rows is not 3/2 of an even height") 

203 return packed[:h], packed[h:]

206def nv12_height(packed: Any) -> int: 

207 """Image height encoded by a contiguous NV12 ``(H + H//2, W)`` buffer.""" 

208 rows = int(_as_cupy(packed).shape[0]) 

209 return rows * 2 // 3

212def cuda_frame( 

213 array: Any, 

214 *, 

215 pixel_format: str = "auto", 

216 width: int | None = None, 

217 height: int | None = None, 

218 seq: int = 0, 

219 timestamp_us: int = 0, 

220) -> RawFrame: 

221 """Wrap a device tensor as a CUDA :class:`~pdum.rfb.types.RawFrame` for ``publish()``. 

222 

223 ``pixel_format="auto"`` infers from shape: ``(H, W, 3)`` -> ``rgb24``, 

224 ``(H, W, 4)`` -> ``rgba8``, 2-D ``(H+H//2, W)`` -> ``nv12``. Pass an explicit 

225 ``pixel_format`` (and ``height`` for ambiguous NV12) to override. The tensor 

226 is referenced, not copied; keep it alive until the frame is encoded. 

227 """ 

228 arr = _as_cupy(array) 

229 if pixel_format == "auto": 

230 if arr.ndim == 3 and arr.shape[2] == 3: 

231 pixel_format = "rgb24" 

232 elif arr.ndim == 3 and arr.shape[2] == 4: 

233 pixel_format = "rgba8" 

234 elif arr.ndim == 2: 

235 pixel_format = "nv12" 

236 else: 

237 raise ValueError(f"cannot infer pixel_format from shape {arr.shape!r}") 

238 if pixel_format == "nv12": 

239 width = int(arr.shape[1]) if width is None else width 

240 height = nv12_height(arr) if height is None else height 

241 else: 

242 height = int(arr.shape[0]) if height is None else height 

243 width = int(arr.shape[1]) if width is None else width 

244 return RawFrame( 

245 seq=seq, 

246 width=int(width), 

247 height=int(height), 

248 timestamp_us=int(timestamp_us), 

249 pixel_format=pixel_format, # type: ignore[arg-type] 

250 memory="cuda", 

251 data=arr, 

252 )

255def to_host_rgb(frame: RawFrame): 

256 """Download a CUDA frame to a contiguous host ``numpy`` **rgb24** array. 

257 

258 Used by the image / CPU H.264 encoders so an image-only (or CPU-fallback) 

259 client still works when the publisher pushes CUDA frames. ``rgba8`` drops 

260 alpha; ``nv12`` is converted to ``rgb24`` on the GPU first. A host frame is 

261 returned (coerced to 3-channel) unchanged. 

262 """ 

263 import numpy as np 

264 

265 if frame.memory == "cpu": 

266 arr = frame.data 

267 return np.ascontiguousarray(arr[:, :, :3]) if getattr(arr, "ndim", 0) == 3 and arr.shape[2] == 4 else arr 

268 

269 import cupy as cp 

270 

271 arr = _as_cupy(frame.data) 

272 if frame.pixel_format in ("rgb24", "rgba8"): 

273 return cp.asnumpy(cp.ascontiguousarray(arr[:, :, :3]) if arr.shape[2] == 4 else arr) 

274 if frame.pixel_format == "nv12": 

275 w, h = frame.width, frame.height 

276 rgb = cp.empty((h, w, 3), cp.uint8) 

277 _, k_nv122rgb, block = _kernels() 

278 k_nv122rgb(_grid(w, h, block), block, (cp.ascontiguousarray(arr), rgb, w, h)) 

279 return cp.asnumpy(rgb) 

280 raise ValueError(f"cannot convert {frame.pixel_format!r} CUDA frame to host rgb")

283class HostFrameAdapter: 

284 """Wrap a host :class:`~pdum.rfb.types.EncoderBackend` so it tolerates CUDA frames. 

285 

286 The image / CPU encoders expect host frames; when the publisher pushes CUDA 

287 frames (``serve(gpu=True)``), an image-transport viewer's encoder is wrapped in 

288 this adapter, which downloads each CUDA frame to host ``rgb24`` 

289 (:func:`to_host_rgb`) before delegating. Host frames pass through untouched, so 

290 the wrapped encoder stays dependency-pure and the existing contract is intact. 

291 """ 

292 

293 __slots__ = ("_inner",) 

294 

295 def __init__(self, inner: Any) -> None: 

296 self._inner = inner 

297 

298 def encode(self, frame: RawFrame, *, force_keyframe: bool = False): 

299 return self._inner.encode(self._to_host(frame), force_keyframe=force_keyframe) 

300 

301 def encode_still(self, frame: RawFrame): 

302 """Download a CUDA frame to host, then delegate the still to the wrapped 

303 encoder (keeps "still after settle" working in ``serve(gpu=True)`` + 

304 image-transport viewers).""" 

305 return self._inner.encode_still(self._to_host(frame)) 

306 

307 @staticmethod 

308 def _to_host(frame: RawFrame) -> RawFrame: 

309 if frame.memory == "cpu": 

310 return frame 

311 import dataclasses 

312 

313 return dataclasses.replace(frame, data=to_host_rgb(frame), memory="cpu", pixel_format="rgb24") 

314 

315 def flush(self): 

316 return self._inner.flush() 

317 

318 def close(self) -> None: 

319 self._inner.close()

325def _selftest_zerocopy_encode(width: int = 256, height: int = 128) -> bool: 

326 """Actually run one zero-copy NVENC encode; ``True`` iff PyAV supports it. 

327 

328 This is the real test for the PyAV 18 capability: on PyAV 17.x it raises at 

329 ``avcodec_open2`` ("hw_frames_ctx must be set"). Side effects are confined to 

330 a tiny encode on the GPU. 

331 """ 

332 try: 

333 import av 

334 import cupy as cp 

335 

336 enable_cuda_context_sharing() 

337 from av.video.frame import CudaContext # noqa: PLC0415 

338 

339 cctx = CudaContext(device_id=0, primary_ctx=True) 

340 nv12 = cp.zeros((height + height // 2, width), cp.uint8) 

341 y, uv = nv12[:height], nv12[height:] 

342 cp.cuda.runtime.deviceSynchronize() 

343 ctx = av.CodecContext.create("h264_nvenc", "w") 

344 ctx.width, ctx.height, ctx.pix_fmt = width, height, "cuda" 

345 from fractions import Fraction # noqa: PLC0415 

346 

347 ctx.time_base = Fraction(1, 30) 

348 ctx.framerate = Fraction(30, 1) 

349 ctx.bit_rate = 1_000_000 

350 ctx.options = {"preset": "p4", "tune": "ll", "rc": "vbr", "bf": "0", "delay": "0"} 

351 out = [] 

352 for s in range(2): 

353 f = av.VideoFrame.from_dlpack( 

354 [y, uv], format="nv12", width=width, height=height, primary_ctx=True, cuda_context=cctx 

355 ) 

356 f.pts = s 

357 f.time_base = Fraction(1, 30) 

358 out += [bytes(p) for p in ctx.encode(f)] 

359 out += [bytes(p) for p in ctx.encode(None)] 

360 return sum(len(p) for p in out) > 0 

361 except Exception: 

362 return False

365@functools.lru_cache(maxsize=1) 

366def cuda_zerocopy_available() -> bool: 

367 """True if the zero-copy CUDA→NVENC path is usable in this process (cached). 

368 

369 Checks, in order: CuPy importable; an NVENC-capable GPU/driver 

370 (:func:`pdum.rfb.encoders.nvenc_cpu.nvenc_cpu_available`); and that PyAV can actually 

371 encode a CUDA frame (PyAV ≥ 18 or a from-source build with the fix). The 

372 self-test opens an NVENC session, so it runs at most once per process. 

373 

374 .. note:: 

375 Call :func:`enable_cuda_context_sharing` **before any CuPy CUDA op** for a 

376 reliable result — if CuPy has already activated the primary context with 

377 the default flags, the shared-context probe (and the encoder) will fail. 

378 """ 

379 if importlib.util.find_spec("cupy") is None: 

380 return False 

381 try: 

382 from .encoders.nvenc_cpu import nvenc_cpu_available 

383 

384 if not nvenc_cpu_available(): 

385 return False 

386 except Exception: 

387 return False 

388 return _selftest_zerocopy_encode()
