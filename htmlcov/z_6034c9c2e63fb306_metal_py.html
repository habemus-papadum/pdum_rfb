1"""Apple Metal / MLX GPU frame helpers — the unified-memory analog of :mod:`pdum.rfb.gpu`. 

2 

3Where :mod:`pdum.rfb.gpu` lets a **CUDA** tensor be encoded by NVENC without a host round-trip, 

4this module lets an **MLX** (Apple Metal, unified-memory) frame be converted **RGB(A) → NV12 on 

5the GPU** with a custom ``mx.fast.metal_kernel`` and handed to the VideoToolbox encoder — instead 

6of the CPU color-conversion pass (~6.6 ms at 1080p, and it pegs a core). On Apple Silicon there is 

7no PCIe upload to eliminate, so the remaining copy (host NV12 → ``CVPixelBuffer``) is negligible 

8(≤2 % of frame time, measured); the win here is moving the *color conversion* onto the GPU. 

9 

10Everything lazy-imports ``mlx`` so ``import pdum.rfb`` never requires it. macOS + MLX only; gate on 

11:func:`mlx_available`. The natural producer is a render kernel that writes an ``(H, W, 4)`` RGBA 

12``mx.array`` (see ``examples/mlx_vt_stream.py``); publish it directly (``display.publish(rgba)`` — 

13an MLX array is recognized as a ``memory="metal"`` frame) or wrap a pre-converted NV12 array with 

14:func:`metal_frame`. 

15"""

32 """True if MLX (Apple Metal) is usable in this process (cached). macOS + ``mlx`` importable.""" 

33 if sys.platform != "darwin" or importlib.util.find_spec("mlx") is None:

43@functools.lru_cache(maxsize=1) 

44def _nv12_kernel(): 

45 """The RGB(A)→NV12 Metal kernel (built once). ``C`` (channels) is a template int so the same 

46 kernel handles ``rgb24`` (3) and ``rgba8`` (4). BT.601 limited range — byte-identical to 

47 :func:`pdum.rfb.gpu.rgb_to_nv12` and the CPU path, so a VideoToolbox stream and a CUDA/NVENC 

48 stream tag color the same way.""" 

49 import mlx.core as mx 

50 

51 return mx.fast.metal_kernel( 

52 name="pdum_rgb_to_nv12", 

53 input_names=["rgb"], 

54 output_names=["out"], 

55 source=""" 

56 uint x = thread_position_in_grid.x; 

57 uint y = thread_position_in_grid.y; 

58 if (x >= W || y >= H) return; 

59 uint ri = (y * W + x) * C; 

60 float r = float(rgb[ri + 0]); 

61 float g = float(rgb[ri + 1]); 

62 float b = float(rgb[ri + 2]); 

63 out[y * W + x] = (uint8_t)clamp(0.257f * r + 0.504f * g + 0.098f * b + 16.0f, 0.0f, 255.0f); 

64 if ((x % 2u == 0u) && (y % 2u == 0u)) { 

65 float U = -0.148f * r - 0.291f * g + 0.439f * b + 128.0f; 

66 float V = 0.439f * r - 0.368f * g - 0.071f * b + 128.0f; 

67 uint uv = W * H + (y / 2u) * W + (x / 2u) * 2u; // contiguous NV12 UV plane 

68 out[uv + 0] = (uint8_t)clamp(U, 0.0f, 255.0f); 

69 out[uv + 1] = (uint8_t)clamp(V, 0.0f, 255.0f); 

70 } 

71 """, 

72 )

75def rgb_to_nv12(rgb: Any): 

76 """Convert a Metal ``rgb24``/``rgba8`` MLX array ``(H, W, 3|4)`` to a contiguous NV12 MLX 

77 array ``(H + H//2, W)`` on the GPU. Returns a **lazy** ``mx.array`` (call :func:`to_host_nv12` 

78 or ``mx.eval`` to materialize). Even dimensions required.""" 

79 import mlx.core as mx 

80 

81 if not isinstance(rgb, mx.array): 

82 rgb = mx.array(rgb) 

83 if rgb.ndim != 3 or rgb.shape[2] < 3: 

84 raise ValueError(f"rgb_to_nv12 expects (H, W, 3|4); got shape {tuple(rgb.shape)!r}") 

85 h, w, c = int(rgb.shape[0]), int(rgb.shape[1]), int(rgb.shape[2]) 

86 if w % 2 or h % 2: 

87 raise ValueError(f"NV12 requires even dimensions; got {w}x{h}") 

88 (out,) = _nv12_kernel()( 

89 inputs=[rgb], 

90 template=[("W", w), ("H", h), ("C", c)], 

91 grid=(w, h, 1), 

92 threadgroup=(16, 16, 1), 

93 output_shapes=[(h + h // 2, w)], 

94 output_dtypes=[mx.uint8], 

95 ) 

96 return out

99def materialize(array: Any) -> None: 

100 """Force a lazy MLX array to compute **on the calling thread**. MLX binds a lazy graph's 

101 nodes to the thread's default stream, so an array built on the publish/loop thread cannot be 

102 evaluated on the session's encode *worker* thread ("no Stream(gpu, 0) in current thread"). 

103 :meth:`Display.publish` calls this for Metal frames so the render is materialized on the loop 

104 thread; the (cheap) NV12 conversion then runs safely on the worker thread over the eager buffer.""" 

105 import mlx.core as mx 

106 

107 mx.eval(array)

110def to_host_nv12(array: Any): 

111 """Evaluate a Metal NV12 MLX array and return a contiguous host ``numpy`` view. Unified memory 

112 makes ``np.asarray`` a near-zero-copy handoff; the VideoToolbox binding then does the small 

113 ``CVPixelBuffer`` copy.""" 

114 import mlx.core as mx 

115 import numpy as np 

116 

117 mx.eval(array) 

118 return np.ascontiguousarray(np.asarray(array))

121def metal_frame( 

122 array: Any, 

123 *, 

124 pixel_format: str = "auto", 

125 width: int | None = None, 

126 height: int | None = None, 

127 seq: int = 0, 

128 timestamp_us: int = 0, 

129) -> RawFrame: 

130 """Wrap an MLX (Metal) array as a ``memory="metal"`` :class:`~pdum.rfb.types.RawFrame` for 

131 ``publish()`` — the Metal analog of :func:`pdum.rfb.gpu.cuda_frame`. 

132 

133 ``pixel_format="auto"`` infers from shape: ``(H, W, 3)`` → ``rgb24``, ``(H, W, 4)`` → ``rgba8``, 

134 2-D ``(H+H//2, W)`` → ``nv12``. Use this for a **pre-converted NV12** array; a plain RGBA array 

135 can be handed straight to ``display.publish()`` (it is recognized as a Metal frame). The array 

136 is referenced, not copied; keep it alive (and evaluated) until it is encoded. 

137 """ 

138 import mlx.core as mx 

139 

140 if not isinstance(array, mx.array): 

141 array = mx.array(array) 

142 shape = tuple(int(s) for s in array.shape) 

143 if pixel_format == "auto": 

144 if len(shape) == 3 and shape[2] == 3: 

145 pixel_format = "rgb24" 

146 elif len(shape) == 3 and shape[2] == 4: 

147 pixel_format = "rgba8" 

148 elif len(shape) == 2: 

149 pixel_format = "nv12" 

150 else: 

151 raise ValueError(f"cannot infer pixel_format from shape {shape!r}") 

152 if pixel_format == "nv12": 

153 width = shape[1] if width is None else width 

154 height = (shape[0] * 2 // 3) if height is None else height 

155 else: 

156 height = shape[0] if height is None else height 

157 width = shape[1] if width is None else width 

158 return RawFrame( 

159 seq=seq, 

160 width=int(width), 

161 height=int(height), 

162 timestamp_us=int(timestamp_us), 

163 pixel_format=pixel_format, # type: ignore[arg-type] 

164 memory="metal", 

165 data=array, 

166 )

169def to_host_rgb(frame: RawFrame): 

170 """Download a Metal frame to a contiguous host ``numpy`` **rgb24** array — used by the image / 

171 CPU encoders so an image-only client still works under ``serve(gpu=True)`` on macOS. ``rgba8`` 

172 drops alpha; ``nv12`` is converted back to RGB (BT.601 limited) on the host.""" 

173 import mlx.core as mx 

174 import numpy as np 

175 

176 arr = frame.data 

177 if frame.memory != "metal": # already host 

178 h = np.asarray(arr) 

179 return np.ascontiguousarray(h[:, :, :3]) if getattr(h, "ndim", 0) == 3 and h.shape[2] == 4 else h 

180 mx.eval(arr) 

181 host = np.asarray(arr) 

182 if frame.pixel_format in ("rgb24", "rgba8"): 

183 return np.ascontiguousarray(host[:, :, :3]) if host.shape[2] == 4 else np.ascontiguousarray(host) 

184 if frame.pixel_format == "nv12": 

185 return _nv12_to_rgb_host(host, frame.width, frame.height) 

186 raise ValueError(f"cannot convert {frame.pixel_format!r} Metal frame to host rgb")

189def _nv12_to_rgb_host(nv12, w: int, h: int): 

190 """BT.601 limited-range NV12 ``(H+H//2, W)`` → host ``rgb24``. The inverse of the encode-side 

191 conversion; only used for the image-transport fallback of a pre-converted NV12 Metal frame.""" 

192 import numpy as np 

193 

194 y = nv12[:h].astype(np.float32) 

195 uv = nv12[h:].reshape(h // 2, w // 2, 2) 

196 u = np.repeat(np.repeat(uv[..., 0], 2, 0), 2, 1).astype(np.float32) - 128.0 

197 v = np.repeat(np.repeat(uv[..., 1], 2, 0), 2, 1).astype(np.float32) - 128.0 

198 c = y - 16.0 

199 r = np.clip(1.164 * c + 1.596 * v, 0, 255) 

200 g = np.clip(1.164 * c - 0.392 * u - 0.813 * v, 0, 255) 

201 b = np.clip(1.164 * c + 2.017 * u, 0, 255) 

202 return np.ascontiguousarray(np.stack([r, g, b], axis=-1).astype(np.uint8))

205def to_host_frame(frame: RawFrame) -> RawFrame: 

206 """Return ``frame`` with its data on the host: a Metal frame is downloaded to a host 

207 ``rgb24`` :class:`~pdum.rfb.types.RawFrame` (:func:`to_host_rgb`); any other frame is 

208 returned unchanged. Lets the CPU / image encoders accept a published MLX frame — the frame 

209 was materialized on the publish thread, so this cross-thread read is safe.""" 

210 if frame.memory != "metal": 

211 return frame 

212 import dataclasses 

213 

214 return dataclasses.replace(frame, data=to_host_rgb(frame), memory="cpu", pixel_format="rgb24")

217class MetalHostFrameAdapter: 

218 """Wrap a host :class:`~pdum.rfb.types.EncoderBackend` so it tolerates Metal frames. 

219 

220 The Metal analog of :class:`pdum.rfb.gpu.HostFrameAdapter`: when the publisher pushes MLX 

221 (Metal) frames under ``serve(gpu=True)`` on macOS, an image-transport viewer's encoder is 

222 wrapped in this adapter, which downloads each Metal frame to host ``rgb24`` 

223 (:func:`to_host_rgb`) before delegating. Host frames pass through untouched. 

224 """ 

225 

226 __slots__ = ("_inner",) 

227 

228 def __init__(self, inner: Any) -> None: 

229 self._inner = inner 

230 

231 def encode(self, frame: RawFrame, *, force_keyframe: bool = False): 

232 return self._inner.encode(self._to_host(frame), force_keyframe=force_keyframe) 

233 

234 def encode_still(self, frame: RawFrame): 

235 return self._inner.encode_still(self._to_host(frame)) 

236 

237 @staticmethod 

238 def _to_host(frame: RawFrame) -> RawFrame: 

239 return to_host_frame(frame) 

240 

241 def flush(self): 

242 return self._inner.flush() 

243 

244 def close(self) -> None: 

245 self._inner.close()
